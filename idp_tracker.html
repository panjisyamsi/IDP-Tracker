// File: idp-tracker.gs // Purpose: Google Apps Script backend untuk IDP Completion Tracker // Sheet: "Status" var SHEET_NAME = "Status"; var COL = { NIK: 1, // B NAME: 2, // C DEPARTMENT: 5, // F SPV: 6, // G NOT_COMPLETED: 12, // M STATUS: 14, // O PROGRESS: 15 // P }; function doGet(e) { var action = e && e.parameter && e.parameter.action ? e.parameter.action : "getData"; var result; try { if (action === "getData") { result = getData(); } else { result = { error: "Unknown action: " + action }; } } catch (err) { result = { error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getData() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { throw new Error("Sheet '" + SHEET_NAME + "' tidak ditemukan. Pastikan nama tab sudah benar."); } var lastRow = sheet.getLastRow(); if (lastRow < 2) { return { employees: [], departments: [] }; } var dataRange = sheet.getRange(2, 1, lastRow - 1, COL.PROGRESS + 1); var values = dataRange.getValues(); var employees = []; var deptSet = {}; for (var i = 0; i < values.length; i++) { var row = values[i]; var nik = String(row[COL.NIK] || "").trim(); var name = String(row[COL.NAME] || "").trim(); var dept = String(row[COL.DEPARTMENT] || "").trim(); var spv = String(row[COL.SPV] || "").trim(); var notCompleted = String(row[COL.NOT_COMPLETED] || "").trim(); var status = String(row[COL.STATUS] || "").trim(); var progressRaw = row[COL.PROGRESS]; if (!nik && !name) continue; var progress = 0; if (typeof progressRaw === "number") { progress = progressRaw <= 1 ? Math.round(progressRaw * 100) : Math.round(progressRaw); } else if (typeof progressRaw === "string") { progress = Math.round(parseFloat(progressRaw.replace("%", "")) || 0); } progress = Math.max(0, Math.min(100, progress)); employees.push({ nik: nik, name: name, dept: dept, spv: spv, notCompleted: notCompleted, status: status, progress: progress }); if (dept) deptSet[dept] = true; } // Sort nama abjad employees.sort(function(a, b) { return a.name.localeCompare(b.name, "id"); }); var departments = Object.keys(deptSet).sort(function(a, b) { return a.localeCompare(b, "id"); }); return { employees: employees, departments: departments, fetchedAt: new Date().toISOString() }; }
